Tutorial: Build a sample 2 view app

Download and install

Download the stage.js distribution. You can either include a script file or use package.json. 

        
# If you use npm use the following command for now
npm install github:naikus/stagejs --save
        
      
        
<!-- Directory structure -->
myapp
  + css
  |  + stage.css
  + js
  |  + stage.js
  + views
  + index.html

Create index.html

Create a file index.html inside myapp directory. Include stage.js and stage.css. 

      
<!DOCTYPE html>
<html>
  <head>
    <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge,chrome=1" />
    <meta name="HandheldFriendly" content="True" />
    <meta name="MobileOptimized" content="480" />
    <meta name="viewport"
        content="width=device-width, initial-scale=1, maximum-scale=1, user-scalable=no" />
    <meta name="apple-mobile-web-app-capable" content="yes" />

    <meta http-equiv="cleartype" content="on" />

    <link rel="stylesheet" type="text/css" href="css/stage.css" />
    <style>
      body {
        font-family: sans-serif;
        font-size: 1em;
      }
    </style>
  </head>

  <!-- Don't keep body as the view-port on android devices cause the views don't scroll -->
  <body>
    <!-- All our views will be shown here -->
    <div class="stage-viewport" id="viewPort"></div>
    <script src="stage.js"> </script>
    <script>
      var viewPort = document.getElementById("viewPort");

      // Define your views
      Stage.views({
        "home": "views/home.html",
        "about": "views/about.html"
      });

      var stage = Stage({ // Or if you are using require , require("stage")
        // viewport: "#viewPort",
        viewport: viewPort
      });

      setTimeout(function() {
        stage.pushView("home");
      }, 100);
    </script>
  </body>
</html>

Create home view partial

Create a file home.html in the views directory. 

        
<!-- views/home.html (partial) -->
<div class="stage-view home" data-view="home">
  <h1>Home</h1>
  <button class="my-button">Show About View</button>
  <!-- Any external script you want to include -->
  <script src="four.js"></script>

  <!-- You can also put the code below in a js file and add a script tag here -->
  <script>
    Stage.defineView("home", function factory(stageContext, viewElement) {
      return {
        initialize: function(options) {
          viewElement.getElementsByClassName("my-button")[0]
              .addEventListener("click", function() {
            var f = four(); // from four.js
            stageContext.showView("about", {f: f});
          }, false);
        },
        activate: function(opts) {},
        deactivate: function(opts) {}
      };
    });
  </script>
</div>

Create about view partial

Create a file about.html in the views directory. 

        
<!-- views/about.html (partial) -->
<div class="stage-view about" data-view="about">
  <h1>About</h1>
  <button class="my-button">Go Back</button>

  <!-- You can also put the code below in a js file and add a script tag here -->
  <script>
    Stage.defineView("about", function factory(stageContext, viewElement) {
      return {
        initialize: function(options) {
          viewElement.getElementsByClassName("my-button")[0]
              .addEventListener("click", function() {
            stageContext.popView();
          }, false);
        }
      };
    });
  </script>
</div>

That's it! Start a server and load index.html from your browser. To use stage with commonjs check out the Stage Statrter App or with Vue.js see Stage Starter Vue App
