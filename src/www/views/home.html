Stage.js

A compact library for simple mobile and web SPAs

Start Here

Introduction

Stage.js is a compact, standalone library that can be used to develop simple, single page mobile and web apps. Stage.js is very focused and provides a view(page) management layer for SPAs. It provides view (page) lifecycle and transitions.

Stage.js does not depend on any external libraries. You are free to use other libraries like jQuery, Zepto, etc. to manage components inside your views. Stage also supports commonJs environment if available. You can also include other libraries like client side router in conjunction with stage.

Usage

Write your view


<!-- views/myview.html (partial) -->
<div class="stage-view myview" data-view="myview">
  <h1>My View</h1>
  <button id="myBtn"></button>
  <!-- Any external script you want to include -->
  <script src="four.js"></script>

  <!-- You can also put the code below in a js file and add a script tag here -->
  <script>
    Stage.defineView("myview", function(stageContext, viewElement) {
      return {
        initialize: function() {
          document.getElementById("myBtn").addEventListener("click", function() {
            var f = four(); // from four.js
            stageContext.showView("about", {f: f});
          }, false);
        },
        activate: function(opts) {},
        deactivate: function(opts) {}
      };
    });
  </script>
</div>

Write your index.html


<!-- index.html -->
<!DOCTYPE html>
<html>
  <head>
    <!-- Include the stage css (or less) -->
    <link rel="stylesheet" type="text/css" href="css/stage.css" />
    <style>
      #viewPort {
        width: 300px;
        height: 450px;
        position: relative;
        margin: 1% auto;
      }
    </style>
  </head>
  <body>
    <div class="stage-viewport" id="viewPort"><div>

    <script src="js/stage.js"> </script>
    <script>
      var viewPort = document.getElementById("viewPort"),
          appStage = Stage({
            // viewport: "#viewPort",
            viewport: viewPort
          });

      // Stage views are registered on the 'Stage' function
      Stage.views({
        "myview": "views/myview.html",
        "about": "views/about.html"
      });

      // You can also register single views
      Stage.view("myView", "views/myview.html");

      setTimeout(function() {
        appStage.pushView("myview" /*, {transition: "slide"}*/);
      }, 500);
    </script>
  </body>
</html>
